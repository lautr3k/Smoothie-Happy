    import uuid from 'uuid/v4'

// Topics collection
const SUBSCRIPTIONS = new Map()

// Publish on topic.
function publish(topic, data = null) {
  let published = 0

  // direct subscription for this topic
  if (SUBSCRIPTIONS.has(topic)) {
    // for each subscription
    SUBSCRIPTIONS.get(topic).forEach(subscription => {
      subscription.callback.call(subscription.context, data, subscription.uuid)
      published++
    })
  }

  return published
}

/**
* PubSub class.
*/
class PubSub {
  /**
  * Subscribe to topic.
  *
  * @param  {String}      topic          Topic name.
  * @param  {Function}    callback       Function to call on topic message.
  * @param  {Object|null} [context=null] Callback context to apply on call.
  * @return {String} Unique identifier (uuid/v4)
  */
  static subscribe(topic, callback, context = null) {
    // first subscription
    if (! SUBSCRIPTIONS.has(topic)) {
      // create new topic
      SUBSCRIPTIONS.set(topic, new Map())
    }

    // create callback wrapper
    let callbackWrapper = { uuid: uuid(), callback, context }

    // register topic callback
    SUBSCRIPTIONS.get(topic).set(callbackWrapper.uuid, callbackWrapper)

    // return the callback uuid
    return callbackWrapper.uuid
  }

  /**
  * Delete entire topic.
  *
  * @param  {String} topic Topic name.
  * @return {Boolean} False if not found
  */
  static deleteTopic(topic) {
    if (SUBSCRIPTIONS.get(topic)) {
      return SUBSCRIPTIONS.delete(topic)
    }

    return false
  }

  /**
  * Unsubscribe from specific topic.
  *
  * @param  {String} uuid Subscription uuid.
  * @return {Boolean} False if not found
  */
  static unsubscribe(uuid) {
    // for each subscription
    for (let topic of SUBSCRIPTIONS.keys()) {
      if (SUBSCRIPTIONS.get(topic).has(uuid)) {
        let result = SUBSCRIPTIONS.get(topic).delete(uuid)

        if (SUBSCRIPTIONS.get(topic).size === 0) {
          this.deleteTopic(topic)
        }

        return result
      }
    }

    return false
  }

  /**
  * Publish on topic.
  *
  * @param {String} topic        Topic name.
  * @param {Mixed}  [data=null]  Topic data.
  * @return {Integer} Number of publications.
  */
  static publish(topic, data = null) {
    // restrict call on catch all
    if (topic[topic.length-1] === '/') {
      throw new Error('You can not publish on catch all topics.')
    }

    let published = 0

    // direct subscription for this topic
    if ((published = publish(topic, data))) {
      // search for catch all
      let topics = []
      let paths  = topic.split('/')

      while (paths.length > 1) {
        paths.pop()
        topics.push(paths.join('/') + '/')
      }

      topics.forEach(name => {
        published += publish(name, { topic: topic.replace(name, ''), data })
      })
    }

    return published
  }
}

// Exports
export default PubSub
export { PubSub }
