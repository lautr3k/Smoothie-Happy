    /**
* @external {ProgressEvent} https://developer.mozilla.org/fr/docs/Web/API/ProgressEvent
*/

/**
* Request event.
*/
class RequestEvent {
  /**
  * @param {String}        type          Event type. See {@link src/http/events/types.js} for possible values.
  * @param {Request}       request       Request instance.
  * @param {ProgressEvent} originalEvent Original progress event.
  */
  constructor(type, request, originalEvent) {
    /**
    * Event type. See {@link src/http/events/types.js} for possible values.
    * @type {String}
    * @protected
    */
    this.type = type

    /**
    * Request instance.
    * @type {Request}
    * @protected
    */
    this.request = request

    /**
    * Original progress event.
    * @type {ProgressEvent}
    * @protected
    */
    this.originalEvent = originalEvent

    /**
    * Progression.
    * @type     {Object|null}
    * @property {Integer} total
    * @property {Integer} loaded
    * @property {Integer} percent
    * @protected
    */
    this.progress = null

    // set progression if computable
    if (originalEvent.lengthComputable) {
      this.progress = {
        total  : originalEvent.total,
        loaded : originalEvent.loaded,
        percent: parseInt(originalEvent.loaded / originalEvent.total * 100)
      }
    }
  }
}

// Exports
export default RequestEvent
export { RequestEvent }
